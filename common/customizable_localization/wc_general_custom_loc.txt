GetCustomUnionName = {
	type = character
	
	text = {
		trigger = {
			is_alive = yes
			exists = var:custom_union_name
			var:custom_union_name = flag:custom
		}
		localization_key = CUSTOM_UNION_NAME
	}
	text = {
		localization_key = "CULTURE_CONFEDERATION"
		fallback = yes
	}
}
